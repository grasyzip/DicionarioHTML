Métodos de Arrays JavaScript

Conheça os principais métodos para manipulação de arrays em JavaScript.

push()
Adiciona um ou mais elementos ao final de um array e retorna o novo comprimento.
array.push(elemento1, elemento2, ...);

Demonstração do método push()

// Array inicial
let frutas = ['maçã', 'banana'];

// Adicionando um elemento
frutas.push('laranja');
console.log(frutas); // ['maçã', 'banana', 'laranja']

// Adicionando múltiplos elementos
let novoComprimento = frutas.push('uva', 'manga');
console.log(frutas); // ['maçã', 'banana', 'laranja', 'uva', 'manga']
console.log('Novo comprimento: ' + novoComprimento); // 5
pop()
Remove o último elemento de um array e retorna esse elemento.
let ultimoElemento = array.pop();

Demonstração do método pop()

// Array inicial
let numeros = [10, 20, 30, 40, 50];

// Removendo o último elemento
let removido = numeros.pop();
console.log('Elemento removido: ' + removido); // 50
console.log('Array após pop: ' + numeros); // [10, 20, 30, 40]

// Removendo mais um elemento
removido = numeros.pop();
console.log('Elemento removido: ' + removido); // 40
console.log('Array após pop: ' + numeros); // [10, 20, 30]
unshift()
Adiciona um ou mais elementos ao início de um array e retorna o novo comprimento.
array.unshift(elemento1, elemento2, ...);

Demonstração do método unshift()

// Array inicial
let cores = ['verde', 'azul'];

// Adicionando um elemento no início
cores.unshift('vermelho');
console.log(cores); // ['vermelho', 'verde', 'azul']

// Adicionando múltiplos elementos no início
let novoComprimento = cores.unshift('amarelo', 'roxo');
console.log(cores); // ['amarelo', 'roxo', 'vermelho', 'verde', 'azul']
console.log('Novo comprimento: ' + novoComprimento); // 5
shift()
Remove o primeiro elemento de um array e retorna esse elemento.
let primeiroElemento = array.shift();

Demonstração do método shift()

// Array inicial
let animais = ['cachorro', 'gato', 'pássaro', 'peixe'];

// Removendo o primeiro elemento
let removido = animais.shift();
console.log('Elemento removido: ' + removido); // 'cachorro'
console.log('Array após shift: ' + animais); // ['gato', 'pássaro', 'peixe']

// Removendo mais um elemento
removido = animais.shift();
console.log('Elemento removido: ' + removido); // 'gato'
console.log('Array após shift: ' + animais); // ['pássaro', 'peixe']
splice()
Adiciona, remove ou substitui elementos em qualquer posição do array.
array.splice(início, quantidade, elemento1, elemento2, ...);

Demonstração do método splice()

// Array inicial
let letras = ['a', 'b', 'c', 'd', 'e'];

// Removendo elementos
let removidos = letras.splice(1, 2); // Remove 2 elementos a partir do índice 1
console.log('Elementos removidos: ' + removidos); // ['b', 'c']
console.log('Array após remoção: ' + letras); // ['a', 'd', 'e']

// Adicionando elementos
letras.splice(1, 0, 'x', 'y'); // Adiciona 'x' e 'y' no índice 1
console.log('Array após adição: ' + letras); // ['a', 'x', 'y', 'd', 'e']

// Substituindo elementos
removidos = letras.splice(2, 2, 'm', 'n'); // Remove 2 elementos do índice 2 e adiciona 'm' e 'n'
console.log('Elementos substituídos: ' + removidos); // ['y', 'd']
console.log('Array após substituição: ' + letras); // ['a', 'x', 'm', 'n', 'e']
Outros Métodos Úteis
Concat, slice, indexOf, includes e outros métodos importantes.
array.concat(outroArray); array.slice(início, fim);

Demonstração de outros métodos úteis

// concat() - Combina arrays
let arr1 = [1, 2];
let arr2 = [3, 4];
let combinado = arr1.concat(arr2);
console.log('Arrays combinados: ' + combinado); // [1, 2, 3, 4]

// slice() - Extrai parte de um array
let numeros = [10, 20, 30, 40, 50];
let parte = numeros.slice(1, 4); // Extrai do índice 1 ao 3
console.log('Parte do array: ' + parte); // [20, 30, 40]

// indexOf() - Encontra o índice de um elemento
let frutas = ['maçã', 'banana', 'laranja'];
let indice = frutas.indexOf('banana');
console.log('Índice de "banana": ' + indice); // 1

// includes() - Verifica se um elemento existe
let existe = frutas.includes('uva');
console.log('"uva" existe no array? ' + existe); // false

// join() - Junta elementos em uma string
let stringFrutas = frutas.join(', ');
console.log('String de frutas: ' + stringFrutas); // "maçã, banana, laranja"
