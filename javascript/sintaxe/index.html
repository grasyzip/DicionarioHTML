Declaração de Variáveis (let, const, var)
Formas de declarar variáveis em JavaScript.
let nome = "João"; const PI = 3.14;

Demonstração de Declaração de Variáveis

// Declaração de variáveis
                            let nome = "Maria";       // Pode ser reatribuída
                            const PI = 3.14159;       // Não pode ser reatribuída
                            var idade = 25;           // Forma antiga (escopo de função)

                            nome = "João";            // Válido
                            // PI = 3.15;             // Erro! Constante não pode ser alterada

                            console.log(nome);        // "João"
                            console.log(PI);          // 3.14159
Variáveis (let, const, var)

Declaração e uso de diferentes tipos de variáveis em JavaScript.

// Usando let (pode ser reatribuída)
let nome = "João";
nome = "Maria"; // Válido

// Usando const (não pode ser reatribuída)
const pi = 3.14159;
// pi = 3.14; // Erro!

// Usando var (escopo de função)
var idade = 30;
Tipos de Dados
Tipos primitivos e objetos em JavaScript.
let texto = "Olá"; let numero = 42;

Demonstração de Tipos de Dados

// Tipos primitivos
                            let texto = "JavaScript";  // String
                            let numero = 42;          // Number
                            let booleano = true;      // Boolean
                            let nulo = null;          // Null
                            let indefinido;           // Undefined
                            let simbolo = Symbol();   // Symbol (ES6)

                            // Objetos
                            let objeto = { nome: "João" };
                            let array = [1, 2, 3];
                            let data = new Date();

                            console.log(typeof texto);    // "string"
                            console.log(typeof numero);   // "number"
                            console.log(typeof nulo);     // "object" (isso é um bug conhecido)
Operadores
Operadores aritméticos, de comparação e lógicos.
let soma = 5 + 3; console.log(5 === "5");

Demonstração de Operadores

// Operadores aritméticos
                            let soma = 5 + 3;       // 8
                            let subtracao = 5 - 3;  // 2
                            let modulo = 10 % 3;    // 1

                            // Operadores de comparação
                            console.log(5 == "5");   // true (igualdade solta)
                            console.log(5 === "5");  // false (igualdade estrita)
                            console.log(5 !== "5");  // true

                            // Operadores lógicos
                            let maiorQue = 10 > 5;  // true
                            let eLogico = true && false; // false
                            let ouLogico = true || false; // true

                            // Operador ternário
                            let idade = 18;
                            let status = idade >= 18 ? "Adulto" : "Menor";
Estruturas de Controle
Condicionais e loops para controle de fluxo.
if (idade >= 18) {...} for (let i = 0; i < 5; i++) {...}

Demonstração de Estruturas de Controle

// Condicional if-else
                            let hora = 14;
                            if (hora < 12) {
                                console.log("Bom dia!");
                            } else if (hora < 18) {
                                console.log("Boa tarde!");
                            } else {
                                console.log("Boa noite!");
                            }

                            // Switch-case
                            let dia = 3;
                            switch(dia) {
                                case 1: console.log("Segunda"); break;
                                case 2: console.log("Terça"); break;
                                default: console.log("Outro dia");
                            }

                            // Loop for
                            for (let i = 0; i < 3; i++) {
                                console.log("Iteração " + i);
                            }

                            // Loop while
                            let contador = 0;
                            while (contador < 3) {
                                console.log("Contador: " + contador);
                                contador++;
                            }
Funções
Declaração e uso de funções em JavaScript.
function saudacao(nome) { return "Olá " + nome; }

Demonstração de Funções

// Declaração de função
                            function soma(a, b) {
                                return a + b;
                            }

                            // Expressão de função
                            const multiplica = function(a, b) {
                                return a * b;
                            };

                            // Arrow function (ES6)
                            const divide = (a, b) => a / b;

                            // Parâmetros padrão
                            function saudacao(nome = "Visitante") {
                                return `Olá, ${nome}!`;
                            }

                            console.log(soma(2, 3));        // 5
                            console.log(multiplica(2, 3));  // 6
                            console.log(saudacao("João"));  // "Olá, João!"
                            console.log(saudacao());        // "Olá, Visitante!"
Escopo
Escopo global, de função e de bloco em JavaScript.
let global = "acessível em qualquer lugar";

Demonstração de Escopo

// Escopo global
                            let global = "Eu sou global";

                            function testeEscopo() {
                                // Escopo de função
                                let funcao = "Eu sou local à função";
                                
                                if (true) {
                                    // Escopo de bloco (com let/const)
                                    let bloco = "Eu sou local ao bloco";
                                    console.log(bloco);  // Acessível
                                }
                                
                                console.log(global);    // Acessível
                                console.log(funcao);    // Acessível
                                // console.log(bloco);  // Erro! Não acessível
                            }

                            testeEscopo();
                            console.log(global);        // Acessível
                            // console.log(funcao);    // Erro! Não acessível
                            // console.log(bloco);     // Erro! Não acessível
Template Literals
Strings com interpolação e multi-linha (ES6).
`Olá ${nome}, hoje é ${new Date()}`

Demonstração de Template Literals

const nome = "Carlos";
                            const idade = 30;

                            // String tradicional
                            let mensagem1 = "Olá " + nome + ", você tem " + idade + " anos.";

                            // Template literal
                            let mensagem2 = `Olá ${nome}, você tem ${idade} anos.`;

                            // Multi-linha
                            let html = `
                            <div class="usuario">
                                <h2>${nome}</h2>
                                <p>Idade: ${idade}</p>
                            </div>
                            `;

                            console.log(mensagem1);
                            console.log(mensagem2);
                            console.log(html);
